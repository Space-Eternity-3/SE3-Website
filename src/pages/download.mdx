# Download page

export const System = () => {
    if (window.navigator.userAgent.indexOf("Windows") != -1) return "Windows";
    if (window.navigator.userAgent.indexOf("Mac") != -1) return "Mac";
    if (window.navigator.userAgent.indexOf("Linux") != -1) return "Linux";
};

export const DownloadLink = () => {
    if (System() === "Windows")
        return "https://github.com/Space-Eternity-3/SE3-Launcher/releases/download/v3.0.0/SE3.Launcher-3.0.0-win-x64.exe";
    if (System() === "Linux")
        return "https://github.com/Space-Eternity-3/SE3-Launcher/releases/download/v3.0.0/SE3.Launcher-3.0.0-linux-x86_64.AppImage";
};

export const GetMessage = () => {
    if (System() === "Windows") return "Download for Windows x64";
    if (System() === "Linux") return "Download for Linux";
    if (System() === "Mac") return "MacOS is unsupported";
};
